### Math

```
class Main {
    public static void main(String[] args) {

        // int addition
        int x = 6;
        int y = 3;
        System.out.println(x+y);

        // float or decimal addition
        double price1 = 24.99;
        double price2 = 19.45;
        double sum = price1 + price2;
        System.out.printl(sum);

        // multiplication
        int price = 1200;
        System.out.printl(price * 0.2);

        // division is run down when int , use double to get result of division with a decimal
        int x = 15; int y = 4;
        int result = x / y;
        System.out.println(result); // 3

        double amount = 9000;
        double price = 49;
        double result = amount/price;
        System.out.print(result);

        // modulo
        int items = 23;
        int res = items % 5; 
        System.out.print(res);

        // addition with string
        String firstname = "James";
        String lastname = "Smith";
        String fullname = firstname+lastname;
        System.out.print(fullname);

    }
}
```
